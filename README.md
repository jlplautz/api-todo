# api-todo
A simple todo list API

## Alguns comandos executados
  - cd PycharmProjects
  - mkdir api-todo
  - cd api-todo
  - pyenv versions
  - pyenv update
  - pyenv install --list
  - pyenv install 3.9.7
  - pyenv install --list
  - pyenv versions
  - pyenv local 3.9.7
  - python -m venv .venv
  - python -m venv .venv/bin/activate
  - pip install poetry
  - poetry version
  - poetry -V
  - poetry self update
  - poetry -V
  - poetry init
  - git init
  - git remote -v
  - git remote add origin git@github.com:jlplautz/api-todo.git 
  - git remote -v
  - ls -la
  - source .venv/bin/activate
  - ls -la
  - git fetch all
  - git fetch --all
  - git pull origin main
  - ls -la
  - git checkout -b 1-startproject
  - git status
  - git add pyproject.toml
  - git status
  - git commit -m 'Started project'
  - git push origin 1-startproject
  - cd PycharmProjects
  - cd api-todo
  - source .venv/bin/activate
  - git checkout main
  - git fetch --all

  - poetry install
